    ->Limpar o projeto

--CONFIGURANDO ESLINT E PRETTIER
  ->Criar '.editorconfig'
  ->yarn add eslint -D
    ->yarn eslint --init
      ->Excluir 'packge-lock.json' e executar comando 'yarn'
      ->Criar arquivo '.eslintrc'
      ->Instalar plugins:
          ->yarn add prettier eslint-config-prettier eslint-plugin-prettier
            babel-eslint -D
      ->Criar arquivo '.prettierrc"
